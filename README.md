## Install

```bash
yarn add avatardomus-components
```

## Usage

```jsx
import React from 'react'

import { ImageMessage } from 'avatardomus-components'

export default function Message() {
  return <ImageMessage type="Ask" text="Do you want to ask a question?" />
}
```
